10.Tocheckwhethernumberis palindromeornot.
#include<stdio.h>
intmain(intargc,char*argv[])
{
intnum,rev=0,digit,orig;
num=atoi(argv[1]);
orig=num;
while(num>0){
digit=num%10;
rev=rev*10+digit;
num=num/10;
}
if(orig==rev)
{
printf("palindrome");
}
else
printf("notpalindrome");
return0;
}
