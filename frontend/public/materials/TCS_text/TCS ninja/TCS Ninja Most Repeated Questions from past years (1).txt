Q1. Write a Program to print whether the given alphabet is vowel or 
consonant 
Solution:  
#include<stdio.h> 
int main() 
{ 
   // Get the character 
 char ch; 
 scanf("%c", &ch); 
 if(ch >= 'A' && ch <= 'Z') 
{ 
 ch = 'a' + (ch - 'A'); 
 } 
 if(ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u') 
 { 
printf("Vowel"); 
} 
 else 
  { 
    printf("Consonant"); 
  } 
  return 0; 
} 
 
Q2. Write a program to Check whether a given number is a prime 
number or not 
Solution: 
#include <stdio.h> 
#include<math.h> 
int main() 
{ 
    int n, i, flag = 0; 
    //printf("Enter a positive integer: "); 
    scanf("%d",&n); 
    for(i=2; i<=sqrt(n); ++i) 
    { 
        // condition for nonprime number 
        if(n%i==0) 
        { 
            flag=1; 
            break; 
        } 
    } 
 
    if (flag==0) 
        printf("%d is a prime number.",n); 
    else 
        printf("%d is not a prime number.",n);  
    return 0; 
} 
 
Q3. Write a program to change the case of the given alphabet and 
print. 
Solution:  
#include <stdio.h> 
int main() { 
 char c; 
 scanf("%c", &c); 
 // Upper to lower case 
 if('A' <= c && c <= 'Z') 
 { 
     printf("%c", 'a' + (c - 'A')); 
 } 
 // Lower to upper case 
 if('a' <= c && c <= 'z') 
 { 
     printf("%c", 'A' + (c - 'a')); 
 } 
 return 0; 
}

void check_sum_and_diplay(int arr[], int size, int sum) 
{ 
    int i,j; 
    for(i=0; i<size-1; i++) 
    { 
        for(j=i+1; j<size; j++) 
        { 
            if(sum == (arr[i] + arr[j])) 
            { 
                printf("Perfect couple: %d %d", arr[i], arr[j]);

            } 
        } 
    }

 if(isAlphabet == 1)
